# macos install guide
# https://formulae.brew.sh/formula/gcc#default
# brew install gcc@11
# similar environmemnt to codeforces GNU G++20 11.2.0

cmake_minimum_required(VERSION 3.19)
set(CMAKE_CXX_COMPILER "/usr/local/Cellar/gcc/11.2.0/bin/g++-11")
project(algo LANGUAGES CXX)
set(CMAKE_CXX_STANDARD 20)
set(CMAKE_RUNTIME_OUTPUT_DIRECTORY ${CMAKE_CURRENT_SOURCE_DIR})
add_executable(algo codeforces.cpp)
